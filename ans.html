                        #define NUM_PHILOSOPHERS 5
                        
                        pthread_mutex_t forks[NUM_PHILOSOPHERS];
                        sem_t semaphore;
                        
                        void *philosopher(void *arg) {
                            int id = *(int *)arg;
                            int left_fork = id;
                            int right_fork = (id + 1) % NUM_PHILOSOPHERS;
                        
                            while (1) {
                                printf("Philosopher %d is thinking.\n", id);
                        
                                sem_wait(&semaphore);
                                pthread_mutex_lock(&forks[left_fork]);
                                pthread_mutex_lock(&forks[right_fork]);
                                sem_post(&semaphore);
                        
                                printf("Philosopher %d is eating.\n", id);
                        
                                pthread_mutex_unlock(&forks[left_fork]);
                                pthread_mutex_unlock(&forks[right_fork]);
                        
                                printf("Philosopher %d is done eating.\n", id);
                        
                                // Sleep to simulate thinking and eating times
                                usleep(rand() % 1000000);
                            }
                        }
                        
                        int main() {
                            pthread_t philosophers[NUM_PHILOSOPHERS];
                            int ids[NUM_PHILOSOPHERS];
                        
                            // Initialize mutexes and semaphore
                            sem_init(&semaphore, 0, NUM_PHILOSOPHERS - 1);
                            for (int i = 0; i < NUM_PHILOSOPHERS; i++) {
                                pthread_mutex_init(&forks[i], NULL);
                            }
                        
                            // Create philosopher threads
                            for (int i = 0; i < NUM_PHILOSOPHERS; i++) {
                                ids[i] = i;
                                pthread_create(&philosophers[i], NULL, philosopher, &ids[i]);
                            }
                        
                            // Wait for philosopher threads to finish (not normally done in a real program)
                            for (int i = 0; i < NUM_PHILOSOPHERS; i++) {
                                pthread_join(philosophers[i], NULL);
                            }
                        
                            // Clean up resources
                            sem_destroy(&semaphore);
                            for (int i = 0; i < NUM_PHILOSOPHERS; i++) {
                                pthread_mutex_destroy(&forks[i]);
                            }
                        
                            return 0;
                        }

                #!/bin/bash

# Function to calculate Fibonacci
fibonacci() {
    if [ "$1" -le 1 ]; then
        echo "$1"
    else
        echo $(( $(fibonacci $(( $1 - 1 ))) + $(fibonacci $(( $1 - 2 ))) ))
    fi
}

# Input: The position of the Fibonacci sequence to calculate
read -p "Enter the position in the Fibonacci sequence: " position

# Check if the input is a non-negative integer
if [[ "$position" =~ ^[0-9]+$ ]]; then
    result=$(fibonacci $position)
    echo "The Fibonacci number at position $position is: $result"
else
    echo "Invalid input. Please enter a non-negative integer."
fi

                #!/bin/bash

# Function to calculate factorial
factorial() {
    if [ "$1" -eq 0 ]; then
        echo 1
    else
        echo $(( $1 * $(factorial $(( $1 - 1 ))) ))
    fi
}

# Input: The number to calculate the factorial
read -p "Enter a number to calculate its factorial: " num

# Check if the input is a non-negative integer
if [[ "$num" =~ ^[0-9]+$ ]]; then
    result=$(factorial $num)
    echo "The factorial of $num is: $result"
else
    echo "Invalid input. Please enter a non-negative integer."
fi

                #!/bin/bash

# Input: File or text
read -p "Enter the file name or input text: " input

# Check if the input is a file
if [ -f "$input" ]; then
    # Count words in the file
    word_count=$(wc -w < "$input")
    echo "The number of words in $input is: $word_count"
else
    # Count words in the input text
    word_count=$(echo "$input" | wc -w)
    echo "The number of words in the input text is: $word_count"
fi
